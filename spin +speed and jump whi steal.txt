local function runDevourLag()
    task.wait(1)
    Lighting.GlobalShadows = false
    Lighting.EnvironmentDiffuseScale = 0
    Lighting.EnvironmentSpecularScale = 0
    for _, v in ipairs(Lighting:GetChildren()) do if v:IsA("PostEffect") then v.Enabled = false end end
    for _, v in ipairs(workspace:GetDescendants()) do if v:IsA("ParticleEmitter") then v.Enabled = false end end

    local function cleanCharacter(char)
        if char == Player.Character then return end
        for _, a in ipairs(char:GetChildren()) do if a:IsA("Accessory") then a:Destroy() end end
        char.ChildAdded:Connect(function(c) if c:IsA("Accessory") then c:Destroy() end end)
    end
    for _, h in ipairs(workspace:GetDescendants()) do if h:IsA("Humanoid") then cleanCharacter(h.Parent) end end

    -- Ladder Logic
    local LADDER_NAME = "Ladder"
    local savedLadder
    local function findLadder()
        for _, v in ipairs(workspace:GetDescendants()) do if v.Name == LADDER_NAME and v:IsA("Model") then return v end end
    end
    local function saveLadder()
        if savedLadder then return end
        local ladder = findLadder()
        if ladder then savedLadder = ladder:Clone(); savedLadder.Parent = nil end
    end
    local function spawnLadder()
        if not savedLadder then return end
        local char = Player.Character
        if not char then return end
        local hrp = char:WaitForChild("HumanoidRootPart", 5)
        if not hrp then return end
        local ladder = savedLadder:Clone()
        ladder.Parent = workspace
        if ladder.PrimaryPart then ladder:SetPrimaryPartCFrame(hrp.CFrame * CFrame.new(0, 0, -7)) end
    end
    saveLadder()
    spawnLadder()

    -- Auto Devour
    local char = Player.Character
    if char then
        local humanoid = char:FindFirstChildOfClass("Humanoid")
        local tool = Backpack:FindFirstChild("Quantum Cloner")
        if humanoid and tool then
            humanoid:EquipTool(tool)
            task.wait(0.1)
            for _, t in ipairs(Backpack:GetChildren()) do if t:IsA("Tool") then t.Parent = char end end
            task.wait(0.1)
            tool:Activate()
        end
    end
end

local function setupSpin(char)
    local hrp = char:WaitForChild("HumanoidRootPart")
    local existing = hrp:FindFirstChild("ElevenSpin")
    if existing then existing:Destroy() end
    local bav = Instance.new("BodyAngularVelocity")
    bav.Name = "ElevenSpin"
    bav.MaxTorque = Vector3.new(0, math.huge, 0)
    bav.P = 100000
    bav.AngularVelocity = Vector3.new(0, 0, 0)
    bav.Parent = hrp
end

Player.CharacterAdded:Connect(setupSpin)
if Player.Character then setupSpin(Player.Character) end

UIS.JumpRequest:Connect(function()
    if not jumpEnabled then return end
    local char = Player.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if hrp then
        local v = hrp.AssemblyLinearVelocity
        hrp.AssemblyLinearVelocity = Vector3.new(v.X, JUMP_VELOCITY, v.Z)
    end
end)

RunService.RenderStepped:Connect(function()
    local char = Player.Character
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local bav = hrp and hrp:FindFirstChild("ElevenSpin")

    if spinning and bav then
        bav.AngularVelocity = Vector3.new(0, spinSpeed, 0)
        hum.AutoRotate = false
    elseif bav then
        bav.AngularVelocity = Vector3.new(0, 0, 0)
        hum.AutoRotate = true
    end

    SpinBtn.Text = "Spinbot: " .. (spinning and "ON" or "OFF")
    SpeedBtn.Text = "Speed: " .. (speedEnabled and "ON" or "OFF")
    JumpBtn.Text = "Jump: " .. (jumpEnabled and "ON" or "OFF")
    MegaphoneBtn.Text = "Megaphone: " .. (megaphoneEnabled and "ON" or "OFF")
    SLockBtn.Text = "Shift Lock: " .. (shiftLockEnabled and "ON" or "OFF")
    AntiRagBtn.Text = "Anti-Ragdoll: " .. (antiRagdollEnabled and "ON" or "OFF")
    
    Player.DevEnableMouseLock = shiftLockEnabled

    if antiRagdollEnabled and hum then
        hum.PlatformStand = false
        hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, false)
        hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
    end
end)

RunService.Heartbeat:Connect(function()
    local char = Player.Character
    if not char or not speedEnabled then return end
    local hrp = char:FindFirstChild("HumanoidRootPart")
    local hum = char:FindFirstChildOfClass("Humanoid")
    if hrp and hum and hum.MoveDirection.Magnitude > 0 then
        hrp.Velocity = Vector3.new(hum.MoveDirection.X * targetSpeed, hrp.Velocity.Y, hum.MoveDirection.Z * targetSpeed)
    end
end)

local function MakeDrag(gui)
    local dragging, dragStart, startPos
    gui.InputBegan:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true; dragStart = input.Position; startPos = gui.Position end end)
    UIS.InputChanged:Connect(function(input) if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then local delta = input.Position - dragStart; gui.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y) end end)
    UIS.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
end
MakeDrag(MainFrame)
